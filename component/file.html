    # 注意

    仅支持一个文件上传，返回File

        <template>
        <div style="display: flex;align-items: center;">
                  <span style="margin-right: 10px">文件上传</span>
                  <ZhbFile
                          :type = false
                        :fileName="fileName"
                        @fileupFunc="getFile"></ZhbFile>
              </div>
    </template>
    <script>
      export default {
        data() {
          return {
            fileName:''
          };
        },
        methods: {
            //获取文件信息
            getFile(data) {
                console.log(data);
            },
        
        },
      };
    </script>

    # 返回文件信息

    {type:this.type,imgFile:this.file}
